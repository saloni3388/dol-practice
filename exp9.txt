import React from "react";
import { useState } from "react";

function Counter1()
{
    const[count,setcount] = useState(0);
    function Increment()
    {
        setcount(count+1);
    }
    function Decrement()
    {
        setcount(count-1);
    }

    return(
        <div>
            {count}
            <br></br>
            <button onClick={Increment}>Increment</button>
            <button onClick={Decrement}>Decrement</button>
        </div>
    );
}
export default Counter1;
